title AMEX ISO8583 Decoder Flow

actor User

User -> AmexDecoder : decode(rawMessage, options)

AmexDecoder -> HeaderDecoder : read MTI + Bitmap
HeaderDecoder --> AmexDecoder : mti + bitmap

AmexDecoder -> MessageDefinitionsRegistry : resolve definition by MTI
MessageDefinitionsRegistry --> AmexDecoder : MessageDefinition

AmexDecoder -> FieldDefinitionAdjuster : apply(options)
FieldDefinitionAdjuster --> AmexDecoder : effective definition

AmexDecoder -> FieldDecoder : decode fields
FieldDecoder --> AmexDecoder : DecodedField(s)

AmexDecoder -> DecodedMessage : build
AmexDecoder -> MessageMapper : map()
MessageMapper --> AmexDecoder : Typed Message

AmexDecoder -> DecodedAmexResult : build result

User -> DecodedPrettyPrinter : print(result)
DecodedPrettyPrinter --> User : formatted output

@enduml
